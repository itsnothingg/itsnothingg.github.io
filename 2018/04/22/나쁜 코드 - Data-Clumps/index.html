게시판을 조회하는 간단한 함수를 만든다고 생각해보자.

// BoardPage1.js
function search(page, title, startDate, endDate, limit) {
 $.get("/board1", {
 page, title, startDate, endDate, limit
 }).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}

파라미터로 조회할 페이지, 검색어, 검색 시작 일자, 검색 종료 일자, 검색 갯수를 받고 그에따라 api를 호출하는 단순한 함수이다. 아무 문제 없이 잘 동작하는 코드이다.

그러다 게시판이 추가되어 BoardPage2에 해당하는 조회 함수를 또 추가해야하는 상황이 생겼다. BoardPage2에는 검색 카테고리가 추가되어야 한다.

// BoardPage2.js
function search(page, title, startDate, endDate, limit, category) {
 $.get("/board2", {
 page, title, startDate, endDate, limit, category
 }).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}

BoardPage1.js 에 있는 함수를 복붙하여 category 파라미터를 추가한 후에 api에 추가로 넘긴다. 깔끔하게 동작한다.

또 다시 요청이 들어와 BoardPage3에 대한 함수를 만들어야 한다. 다만 BoardPage3의 검색에는 특수문자가 들어갈 수도 있고 검색 시작일자와 끝나는 날자가 유효한 날자 자료형인지 검사해야 한다고 한다. 역시 일단 BoardPage1.js 에 있는 함수를 복붙하여 기본 틀을 만들어 놓고, 필요한 부분을 수정하기로 한다.

// BoardPage3.js
function search(page, title, startDate, endDate, limit) {
 if (!isDate(startDate)) return
 if (!isDate(endDate)) return
 title = escape(title)
 $.get("/board3", {
 page, title, startDate, endDate, limit, category
 }).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}

우리는 복붙의 달인이기 때문에 게시판이 늘어날 때마다 BoardPage1.js 파일을 열어 조회 함수를 복사하여 그대로 가져다 쓰고 필요한 부분만 약간씩 수정하여 사용할 것이다. 중복되는 코드들이 여기 저기 쌓이는 것 같지만, 일단 게시판을 빨리 붙히는 게 중요하니 신경쓰지 않기로 한 채 말이다.

이런 “복붙” 유형의 코드들은 소스의 부피를 키울 뿐만 아니라, 가독성을 심하게 떨어뜨리고, 수정에 무방비하다. 예를 들어 위 게시판 조회 함수에서, 전 페이지에 공통적으로 검색어 제한, 검색어 길이 제한, 날짜 제한 등과 같은 검증을 진행하는 부분을 추가해야한다고 하면, 현재 구조로는 이 검증을 모든 조회 함수에 집어 넣게 될 것이다. 마찬가지로 이 검증에 수정 사항이 발생했을 때도 다시 모든 함수를 찾아다니며 고쳐야한다. 이렇게 변수의 집합이나 코드들이 계속해서 뭉쳐다니는 것을 DataClump라고 부른다.

코드에서 DataClump를 제거하기 위해서는 반복되는 부분들을 공통화 시킬 필요가 있다. 새 클래스를 작성하여 반복되는 코드를 추출하는 것이 그 중 한가지 방법이다. 예를들어 우리가 작성하였던 3개의 게시판은 아래와 같이 재구성 될 수 있다.

// PageParameter.js
class PageParameter {
 page = 1
 title = ''
 startDate = ''
 limit = ''
 endDate = ''
 constructor(page, title, startDate, endDate, limit) {
 this.page = page
 this.title = escape(title)
 this.startDate = new Date(startDate)
 this.endDate = new Date(endDate)
 this.limit = limit
 }
 valueOf() {
 return {this.page, this.title, this.stateDate, this.endDate, this.limit}
 }
}
// BoardPage1.js
function search(pageParam) {
$.get("/board1", pageParam).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}
// BoardPage2.js
function search(pageParam, category) {
$.get("/board2", {
 …pageParam, category
 }).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}
// BoardPage3.js
function search(pageParam) {
 $.get("/board3", pageParam).done(function( data ) {
 alert( "Data Loaded: " + data )
 });
}

PageParameter 클래스를 선언하여 반복되는 변수들을 묶었다. 이렇게 DataClumps를 제거 했을 때의 장점은 코드가 훨씬 조직되어 이해하기 쉬워지고 수정이나 오류가 생겼을 때도 공통화된 클래스만 수정하면 모든 곳에 수정이 반영되어 훨씬 효율적이다.
